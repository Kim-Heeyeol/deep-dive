    x:1,
    foo(){ return this.x}
    bar: function(){return this.x}
}

console.log(obj.bar())  //1.     -(b)

new obj.bar() // -(d)

  name: "Seo",
  sayHi() {
    return `Hi! ${this.name}`;
  },
};

const dervied = {
  __proto__: base,
  sayHi() {
    return `${super.work()}. How are you doing?`;
  },
};

console.log(derived.sayHi()); //(a)

  __proto__: base,
